\subsection{RSA Algorithm}

\begin{frame}
\frametitle{RSA Algorithm: The Challenge}
\begin{itemize}
    \item \textbf{RSA Security}: Based on integer factorization
    \item \textbf{Classical Approach}:
    \begin{itemize}
        \item Best known algorithm: General Number Field Sieve (GNFS)
        \item Time complexity: $O(e^{c(\log N)^{1/3}(\log \log N)^{2/3}})$
        \item 2048-bit RSA: Practically secure against classical computers
    \end{itemize}
    \item \textbf{Quantum Threat}: Shor's Algorithm
    \begin{itemize}
        \item Time complexity: $O((\log N)^3)$
        \item Polynomial time factorization
        \item Threatens current RSA encryption
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Shor's Algorithm: Quantum Solution}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{How Shor's Algorithm Works}
\begin{enumerate}
    \item Quantum Fourier Transform
    \item Period finding subroutine
    \item Factorization via period
    \item Polynomial time solution
\end{enumerate}
\end{column}
\begin{column}{0.5\textwidth}
\textbf{Impact}
\begin{itemize}
    \item Breaks RSA with sufficient qubits
    \item Requires $\sim$4000 logical qubits
    \item Currently: NISQ limitations
    \item Future: Post-quantum cryptography needed
\end{itemize}
\vspace{0.3cm}
\textbf{Status}: 
\begin{itemize}
    \item Theoretical: Proven
    \item Practical: Awaiting fault-tolerant quantum computers
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\subsection{Quantum Key Distribution}

\begin{frame}
\frametitle{Quantum Key Distribution (QKD)}
\begin{itemize}
    \item \textbf{Purpose}: Secure key exchange using quantum mechanics
    \item \textbf{Principle}: Heisenberg Uncertainty Principle
    \begin{itemize}
        \item Measurement disturbs quantum state
        \item Eavesdropping is detectable
        \item Information-theoretic security
    \end{itemize}
    \item \textbf{Advantages over Classical}:
    \begin{itemize}
        \item Security based on physics, not computational difficulty
        \item Detects interception
        \item Future-proof against quantum computers
    \end{itemize}
\end{itemize}
\end{frame}

\subsubsection{BB84 Protocol}

\begin{frame}
\frametitle{BB84 Protocol: Quantum Key Distribution}
\begin{itemize}
    \item \textbf{Invented}: 1984 by Bennett and Brassard
    \item \textbf{Key Components}:
    \begin{enumerate}
        \item \textbf{Qubit Preparation}: Alice sends qubits in random bases
        \begin{itemize}
            \item Basis 1: $\{|0\rangle, |1\rangle\}$ (Z-basis)
            \item Basis 2: $\{|+\rangle, |-\rangle\}$ (X-basis)
        \end{itemize}
        \item \textbf{Qubit Measurement}: Bob measures in random bases
        \item \textbf{Key Sifting}: Public discussion of bases
        \item \textbf{Error Estimation}: Check for eavesdropping
        \item \textbf{Privacy Amplification}: Final secure key
    \end{enumerate}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{BB84 Protocol: Security Features}

    \item Integration with quantum networks
\end{itemize}
\end{frame}
